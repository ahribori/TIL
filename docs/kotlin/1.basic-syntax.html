# Auto type casting (자동 형 변환)

fun getStringLength(obj: Any): Int? {
    if (obj is String) {
        // obj가 자동으로 String으로 변환 됨
        return obj.length
    }
    return null
}

fun describe(obj: Any) : String =
    when (obj) {
        1           -> "One"
        "Hello"     -> "Greeting"
        is Long     -> "Long"
        !is Long    -> "Not a Long"
        else        -> "Unknown"
